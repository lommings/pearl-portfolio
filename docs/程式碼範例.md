## 影片播放器組件

### 自動切換 YouTube/Bilibili

`src/components/VideoPlayer.astro`：

```astro
---
interface Props {
  youtube?: string;
  bilibili?: string;
  title: string;
  autoSwitch?: boolean; // 是否自動切換
}

const { youtube, bilibili, title, autoSwitch = true } = Astro.props;
---

<div class="video-player">
  {youtube && (
    <div class="video-frame youtube" data-platform="youtube">
      <iframe
        src={`https://www.youtube.com/embed/${youtube}`}
        title={title}
        frameborder="0"
        allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"
        allowfullscreen
        loading="lazy"
      ></iframe>
    </div>
  )}
  
  {bilibili && (
    <div class="video-frame bilibili" data-platform="bilibili">
      <iframe
        src={`https://player.bilibili.com/player.html?bvid=${bilibili}&high_quality=1`}
        title={title}
        frameborder="0"
        scrolling="no"
        allowfullscreen
        loading="lazy"
      ></iframe>
    </div>
  )}
  
  {youtube && bilibili && (
    <div class="platform-switcher">
      <button data-switch="youtube">YouTube</button>
      <button data-switch="bilibili">Bilibili</button>
    </div>
  )}
</div>

<script define:vars={{ autoSwitch, youtube, bilibili }}>
  if (autoSwitch && youtube && bilibili) {
    // 簡易地區偵測（根據時區）
    const offset = new Date().getTimezoneOffset();
    const isChinaTime = offset === -480; // UTC+8
    
    const youtubeFrame = document.querySelector('[data-platform="youtube"]');
    const biliFrame = document.querySelector('[data-platform="bilibili"]');
    
    if (isChinaTime) {
      youtubeFrame.style.display = 'none';
      biliFrame.style.display = 'block';
    } else {
      youtubeFrame.style.display = 'block';
      biliFrame.style.display = 'none';
    }
  }
  
  // 手動切換按鈕
  document.querySelectorAll('[data-switch]').forEach(btn => {
    btn.addEventListener('click', (e) => {
      const platform = e.target.dataset.switch;
      document.querySelectorAll('.video-frame').forEach(frame => {
        frame.style.display = frame.dataset.platform === platform ? 'block' : 'none';
      });
      
      // 按鈕狀態
      document.querySelectorAll('[data-switch]').forEach(b => {
        b.classList.toggle('active', b.dataset.switch === platform);
      });
    });
  });
</script>

<style>

  }

    padding-bottom: 56.25%; /* 16:9 */

  }
```

## 遊戲嵌入組件

### 懶載入遊戲 iframe

`src/components/GameEmbed.astro`：

```astro
---
interface Props {
  src: string;
  title: string;
  width?: number;
  height?: number;
}

const { src, title, width = 800, height = 600 } = Astro.props;
---

<div class="game-embed" data-src={src} data-title={title}>
  <div class="game-placeholder">
    <button class="load-game-btn">
      🎮 點擊載入遊戲
    </button>
    <p>{title}</p>
  </div>
</div>

<script>
  document.querySelectorAll('.load-game-btn').forEach(btn => {
    btn.addEventListener('click', function() {
      const container = this.closest('.game-embed');
      const src = container.dataset.src;
      const title = container.dataset.title;
      
      // 建立 iframe
      const iframe = document.createElement('iframe');
      iframe.src = src;
      iframe.title = title;
      iframe.frameBorder = '0';
      iframe.allowFullscreen = true;
      iframe.style.width = '100%';
      iframe.style.height = '600px';
      iframe.style.border = 'none';
      iframe.style.borderRadius = '12px';
      
      // 替換佔位符
      container.innerHTML = '';
      container.appendChild(iframe);
      
      // 全螢幕按鈕
      const fullscreenBtn = document.createElement('button');
      fullscreenBtn.textContent = '🔳 全螢幕';
      fullscreenBtn.className = 'fullscreen-btn';
      fullscreenBtn.onclick = () => {
        if (iframe.requestFullscreen) {
          iframe.requestFullscreen();
        }
      };
      container.appendChild(fullscreenBtn);
    });
  });
</script>
```

```astro
  let currentIndex = 0;
  const images = Array.from(document.querySelectorAll('.gallery-item img'));
  
  // 點擊圖片開啟燈箱

      currentIndex = index;
      showLightbox();
    });
  });
  
  function showLightbox() {
    const lightbox = document.getElementById('lightbox');
    const img = document.getElementById('lightbox-img');
    img.src = images[currentIndex].src;
    img.alt = images[currentIndex].alt;
    lightbox.classList.add('active');

    document.getElementById('lightbox').classList.remove('active');
  });
  
  // 上一張
  document.querySelector('.prev-btn').addEventListener('click', () => {
    currentIndex = (currentIndex - 1 + images.length) % images.length;
    showLightbox();
  });
  
  // 下一張
  document.querySelector('.next-btn').addEventListener('click', () => {
    currentIndex = (currentIndex + 1) % images.length;
    showLightbox();
  });
  
  // ESC 鍵關閉
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') {
      document.getElementById('lightbox').classList.remove('active');
    }
  });
```

## 標籤系統

### 標籤頁面生成

`src/pages/tags/[tag].astro`：

```astro
---
import { getCollection } from 'astro:content';

export async function getStaticPaths() {
  const posts = await getCollection('blog');
  const tags = [...new Set(posts.flatMap(post => post.data.tags || []))];
  
  return tags.map(tag => ({
    params: { tag },
    props: {
      posts: posts.filter(post => post.data.tags?.includes(tag))
    }
  }));
}

const { tag } = Astro.params;
const { posts } = Astro.props;
---

<html>
  <head>
    <title>標籤: {tag}</title>
  </head>
  <body>
    <h1>#{tag}</h1>
    <p>共 {posts.length} 篇文章</p>
    
    <div class="post-list">
      {posts.map(post => (
        <article>
          <h2>
            <a href={`/blog/${post.slug}`}>{post.data.title}</a>
          </h2>
          <p>{post.data.description}</p>
        </article>
      ))}
    </div>
  </body>
</html>
```

## 搜尋功能

### 使用 Pagefind

安裝：

```bash
npm install -D pagefind
```

`astro.config.mjs`：

```javascript
import { defineConfig } from 'astro/config';

export default defineConfig({
  // ...其他設定
  build: {
    // Pagefind 需要
    format: 'directory',
  },
});
```

`package.json` 修改建置指令：

```json
{
  "scripts": {
    "build": "astro build && npx pagefind --source dist"
  }
}
```

搜尋組件 `src/components/Search.astro`：

```astro
<div id="search">
  <input type="search" id="search-input" placeholder="搜尋文章..." />
  <div id="search-results"></div>
</div>

<script>
  import * as pagefind from '/pagefind/pagefind.js';
  
  const searchInput = document.getElementById('search-input');
  const searchResults = document.getElementById('search-results');
  
  searchInput.addEventListener('input', async (e) => {
    const query = e.target.value;
    
    if (query.length < 2) {
      searchResults.innerHTML = '';
      return;
    }
    
    const search = await pagefind.search(query);
    
    searchResults.innerHTML = search.results
      .slice(0, 5)
      .map(result => {
        return `
          <a href="${result.url}">
            <h3>${result.meta.title}</h3>
            <p>${result.excerpt}</p>
          </a>
        `;
      })
      .join('');
  });
</script>
```

```astro
  // 讀取儲存的主題
  const theme = localStorage.getItem('theme') || 
    (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light');
  
  document.documentElement.setAttribute('data-theme', theme);
  
  // 切換主題

    document.documentElement.setAttribute('data-theme', next);
    localStorage.setItem('theme', next);
  });
</script>
```

## 會員系統（Supabase）

### 設定 Supabase

1. 註冊：https://supabase.com
2. 建立新專案
3. 取得 API Key

安裝依賴：

```bash
npm install @supabase/supabase-js
```

### Auth 工具

`src/lib/supabase.ts`：

```typescript
import { createClient } from '@supabase/supabase-js';

const supabaseUrl = import.meta.env.PUBLIC_SUPABASE_URL;
const supabaseKey = import.meta.env.PUBLIC_SUPABASE_ANON_KEY;

export const supabase = createClient(supabaseUrl, supabaseKey);

// 檢查會員資格
export async function checkMembership() {
  const { data: { user } } = await supabase.auth.getUser();
  
  if (!user) return false;
  
  const { data } = await supabase
    .from('memberships')
    .select('*')
    .eq('user_id', user.id)
    .eq('is_active', true)
    .single();
  
  return !!data;
}
```
